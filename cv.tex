\aside{about}{0.03}{
    \footnotesize
    \mbox{I am a PhD candidate at the}
    \mbox{University~of~Washington,}
    \mbox{using simulations to }
    \mbox{understand galaxy evolution.}
    \mbox{I am broadly interested in }
    \mbox{1) the galactic baryon cycle}
    \mbox{and the structure of the}
    \mbox{circumgalactic medium,}
    \mbox{2) how cosmic rays interact }
    \mbox{with gas in various regimes }
    \mbox{and their impact on for galaxy }
    \mbox{evolution, and 3) improving}
    \mbox{existing models of the }
    \mbox{subgrid physics that govern }
    \mbox{galaxy evolution in simulations.}
}
